cmake_minimum_required(VERSION 3.2)

PROJECT(luauuid)
INCLUDE(cmake/ModuleImport.cmake)
INCLUDE(cmake/LuaModule.cmake)
LIST(APPEND CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmake)
INCLUDE(cmake/ModuleCompileOptions.cmake)
ModuleSetCompileOptions()
ModuleSetWinCompilerFlags()

SET(CMAKE_CXX_STANDARD 17)
SET(CMAKE_CXX_STANDARD_REQUIRED ON)

SET(LUA_MODULE lua54)
SET(LUAUUID_VERSION_MAJOR "1")
SET(LUAUUID_VERSION_MINOR "0")
SET(LUAUUID_VERSION_PATCH "1")
SET(LUAUUID_VERSION "${LUAUUID_VERSION_MAJOR}.${LUAUUID_VERSION_MINOR}.${LUAUUID_VERSION_PATCH}")

MESSAGE(STATUS "VERSION: ${LUAUUID_VERSION}")

ADD_DEFINITIONS(-DUSE_LUAUUID)
ADD_DEFINITIONS(-DLUA_COMPAT_MODULE -DLUA_COMPAT_APIINTCASTS)

LuaModuleImport("${LUA_MODULE}" "luauuid" "src/luauuid" "")

IF (WIN32)
ADD_CUSTOM_COMMAND(
    TARGET luauuid
    PRE_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    ${CMAKE_CURRENT_SOURCE_DIR}/config/lua_dkm_debug.json
    $<TARGET_FILE_DIR:luauuid>
)
ENDIF(WIN32)
